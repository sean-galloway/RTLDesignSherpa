```bash
    projects/components/hive/docs/hive_spec/hive_index.md \
    -o projects/components/hive/docs/HIVE_Specification_v0.25.docx \
    --toc --title-page --pdf
```

```bash
pytest val/hive/test_serv_wrapper.py -v
```
